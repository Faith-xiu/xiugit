git add                  
添加到暂存区

git commit            
提交到仓库   
-m '注释'

git status
检查是否提交

git diff
查看修改了什么内容

git log
查看历史记录
--pretty=oneline 查看简化记录

git reset --hard HEAD^
退回到上个版本
HEAD^^  退回到上上个版本
HEAD~*   退回到*个版本

cat 
查看内容

git reset --hard [版本号]
恢复某个版本

git reflog
查看版本号

git restore
撤销修改

rm 
删除文件

git remot add origin [远程库地址]
添加远程库

git remote
查看远程库

git fetch
同步远程库中的数据

git merge
获取远程库中的数据

git push [库名] master
推送到远程库中

find . -name ".git" | xargs rm -Rf
删除本地仓库
